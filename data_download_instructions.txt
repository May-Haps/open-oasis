# 1. Make data directory and cd to it
mkdir -p data/raw && cd data/raw

sudo apt update
sudo apt install -y aria2

# 2. Download Treechop (1.5 GB)
aria2c -x 16 -s 16 -c https://zenodo.org/records/12659939/files/MineRLTreechop-v0.zip

# 3. Verify integrity
echo "f7c0c4e6708c518a0eaf98b48f353661  MineRLTreechop-v0.zip" | md5sum -c

# 4. Unzip
unzip MineRLTreechop-v0.zip
# Creates: MineRLTreechop-v0/<episode_id>/{recording.mp4, rendered.npz, metadata.json}

# 5. (optional) Delete the zip to reclaim 1.5 GB
rm MineRLTreechop-v0.zip

# 6. Check what you got
ls MineRLTreechop-v0/ | head -5
ls MineRLTreechop-v0/$(ls MineRLTreechop-v0/ | head -1)/
# should show: recording.mp4  rendered.npz  metadata.json

# 7. Back to the repo root
cd ../..

# Create train validation split
cd data/raw/MineRLTreechop-v0

# list all episode folders
ls -d */ | tr -d '/' | sort > /tmp/all_episodes.txt
shuf --random-source=<(yes 42) /tmp/all_episodes.txt > /tmp/shuffled.txt

TOTAL=$(wc -l < /tmp/shuffled.txt)
N_VAL=$((TOTAL * 15 / 100))

head -n $N_VAL /tmp/shuffled.txt > /tmp/val_episodes.txt
tail -n +$((N_VAL + 1)) /tmp/shuffled.txt > /tmp/train_episodes.txt

# Create split dirs with symlinks (no copying, no extra disk)
mkdir -p ../treechop_train ../treechop_val

while read ep; do
    ln -s "$PWD/$ep" ../treechop_train/"$ep"
done < /tmp/train_episodes.txt

while read ep; do
    ln -s "$PWD/$ep" ../treechop_val/"$ep"
done < /tmp/val_episodes.txt

echo "Train: $(ls ../treechop_train | wc -l) episodes"
echo "Val:   $(ls ../treechop_val | wc -l) episodes"

cd ../../..   # back to repo root

# Data processing
python data_utils.py \
    --input-dir data/raw/treechop_train \
    --output-dir data/processed_treechop_train \
    --vae-ckpt models/vit-l-20.safetensors \
    --device cuda:0 \
    --batch-size 16

python data_utils.py \
    --input-dir data/raw/treechop_val \
    --output-dir data/processed_treechop_val \
    --vae-ckpt models/vit-l-20.safetensors \
    --device cuda:0 \
    --batch-size 16
